      def print_command_list
        puts "#{Mongrel::Command::BANNER}\nAvailable commands are:\n\n"

        self.commands.each do |name|
          if /mongrel::/ =~ name
            name = name[9 .. -1]
          end

          puts " - #{name[1 .. -1]}\n"
        end

        puts "\nEach command takes -h as an option to get help."

      end
