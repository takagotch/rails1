    def reset!
      logoff rescue nil
      begin
        @connection = @factory.new_connection @username, @password, @database, @async, @prefetch_rows, @cursor_sharing
        __setobj__ @connection
        @active = true
      rescue
        @active = false
        raise
      end
    end
