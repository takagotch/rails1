        def compare_versions(v1, v2)
          v1_seg  = v1.split(".")
          v2_seg  = v2.split(".")
          0.upto([v1_seg.length,v2_seg.length].min) do |i|
            step  = (v1_seg[i].to_i <=> v2_seg[i].to_i)
            return step unless step == 0
          end
          return v1_seg.length <=> v2_seg.length
        end
