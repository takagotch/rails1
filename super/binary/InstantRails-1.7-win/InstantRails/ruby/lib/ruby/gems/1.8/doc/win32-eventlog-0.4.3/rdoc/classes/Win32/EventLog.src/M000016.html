      def total_records
         total = [0].pack('L')
         
         unless GetNumberOfEventLogRecords(@handle, total)
            error = 'GetNumberOfEventLogRecords() failed: '
            error += get_last_error
            raise EventLogError, error
         end
         
         total.unpack('L').first
      end
