      def self.beep(frequency, duration)
         if frequency > HIGH_FREQUENCY || frequency < LOW_FREQUENCY
            raise SoundError, 'invalid frequency'
         end
         
         if 0 == @@Beep.call(frequency, duration)
            raise SoundError, get_last_error
         end        
         self
      end
