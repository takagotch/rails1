   def ftype
      return 'directory' if directory?
      case @file_type
         when FILE_TYPE_CHAR
            'characterSpecial'
         when FILE_TYPE_DISK
            'file'
         when FILE_TYPE_PIPE
            'socket'
         else
            if blockdev?
               'blockSpecial'
            else
               'unknown'
            end
      end
   end
