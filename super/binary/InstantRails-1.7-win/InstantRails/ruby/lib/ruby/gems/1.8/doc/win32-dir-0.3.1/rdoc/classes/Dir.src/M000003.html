   def self.junction?(path)
      bool   = true
      attrib = GetFileAttributes(path)
      
      bool = false if attrib == INVALID_FILE_ATTRIBUTES
      bool = false if attrib & FILE_ATTRIBUTE_DIRECTORY == 0
      bool = false if attrib & FILE_ATTRIBUTE_REPARSE_POINT == 0
      
      bool
   end
