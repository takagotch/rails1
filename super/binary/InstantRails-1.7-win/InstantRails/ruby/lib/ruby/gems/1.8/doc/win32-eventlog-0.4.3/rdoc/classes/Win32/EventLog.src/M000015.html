      def oldest_record_number
         rec = [0].pack('L')
         
         unless GetOldestEventLogRecord(@handle, rec)
            error = 'GetOldestEventLogRecord() failed: ' + get_last_error
            raise EventLogError, error
         end
         
         rec.unpack('L').first
      end
