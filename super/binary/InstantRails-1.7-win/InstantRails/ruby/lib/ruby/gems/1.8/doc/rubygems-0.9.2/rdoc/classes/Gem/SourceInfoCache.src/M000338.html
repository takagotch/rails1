  def cache_data
    return @cache_data if @cache_data
    @dirty = false
    cache_file # HACK writable check
    # Marshal loads 30-40% faster from a String, and 2MB on 20061116 is small
    begin
      data = File.open cache_file, 'rb' do |fp|
        fp.read
      end
      @cache_data = Marshal.load data 
    rescue
      {}
    end
  end
