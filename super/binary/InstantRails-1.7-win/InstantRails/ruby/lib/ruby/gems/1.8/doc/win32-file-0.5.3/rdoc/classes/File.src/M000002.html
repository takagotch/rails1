   def self.securities(mask)
      sec_array = []
      if mask == 0
         sec_array.push('NONE')
      else
         if (mask & FULL) ^ FULL == 0
            sec_array.push('FULL')
         else
            SECURITY_RIGHTS.each{ |string, numeric|
               if (numeric & mask) ^ numeric == 0
                  sec_array.push(string)
               end
            }
         end
      end
      sec_array
   end
